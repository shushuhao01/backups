### 2. 路由配置 ✅

**文件**: `admin/src/router/index.ts`

**新增路由**:
```typescript
{
  path: 'licenses',
  name: 'LicensesParent',
  meta: { title: '授权管理', icon: 'Key' },
  redirect: '/licenses/list',
  children: [
    {
      path: 'list',
      name: 'Licenses',
      component: () => import('@/views/licenses/List.vue'),
      meta: { title: '授权列表' }
    },
    {
      path: 'generate',
      name: 'LicenseGenerate',
      component: () => import('@/views/licenses/Generate.vue'),
      meta: { title: '生成授权' }
    },
    {
      path: 'detail/:id',
      name: 'LicenseDetail',
      component: () => import('@/views/licenses/Detail.vue'),
      meta: { title: '授权详情', hidden: true }
    }
  ]
}
```

**路由地址**:
- 授权列表: `/licenses/list`
- 生成授权: `/licenses/generate`
- 授权详情: `/licenses/detail/:id`

### 4. API接口配置 ✅

**文件**: `admin/src/api/admin.ts`

**已配置的API**:
```typescript
// 授权管理
getLicenses: (params: any) => request.get('/licenses', { params })
getLicense: (id: string) => request.get(`/licenses/${id}`)
createLicense: (data: any) => request.post('/licenses', data)
updateLicense: (id: string, data: any) => request.put(`/licenses/${id}`, data)
deleteLicense: (id: string) => request.delete(`/licenses/${id}`)
activateLicense: (id: string, machineId?: string) => request.post(`/licenses/${id}/activate`, { machineId })
deactivateLicense: (id: string, reason?: string) => request.post(`/licenses/${id}/deactivate`, { reason })
renewLicense: (id: string, expiresAt: string) => request.post(`/licenses/${id}/renew`, { expiresAt })
verifyLicense: (licenseKey: string, machineId?: string) => request.post('/licenses/verify', { licenseKey, machineId })
getLicenseLogs: (id: string, params: any) => request.get(`/licenses/${id}/logs`, { params })
getLicenseStatistics: () => request.get('/licenses/statistics')
```
