  SUBSTRING(MD5(RAND()), 1, 5), '-',
  SUBSTRING(MD5(RAND()), 1, 5)
);

-- 5.2 更新租户状态
UPDATE tenants t
JOIN payment_orders o ON t.id = o.tenant_id
JOIN packages p ON o.package_id = p.id
SET 
  t.status = 'active',
  t.license_key = @license_key,
  t.license_status = 'active',
  t.activated_at = NOW(),
  t.expire_date = DATE_ADD(CURDATE(), INTERVAL p.duration_days DAY),
  t.max_users = p.max_users,
  t.max_storage_gb = p.max_storage_gb,
  t.updated_at = NOW()
WHERE t.id = 'test-tenant-001';

-- 5.3 创建授权记录
INSERT INTO licenses (
  id, license_key, customer_name, customer_type, tenant_id,
  license_type, max_users, max_storage_gb, features,
  status, activated_at, expires_at, created_at, updated_at
)
SELECT 
  UUID(),
  @license_key,
  t.name,
  'tenant',
  t.id,
  'monthly',
  p.max_users,

  COALESCE(p.features, '[]'),
  'active',
  NOW(),
  DATE_ADD(NOW(), INTERVAL p.duration_days DAY),
  NOW(),
  NOW()
FROM tenants t
JOIN payment_orders o ON t.id = o.tenant_id
JOIN packages p ON o.package_id = p.id

## 🐛 常见问题

### 问题1: 租户状态未更新

**原因**: 支付回调未触发激活逻辑

**解决方案**:
1. 检查订单状态是否为 'paid'
2. 检查 tenant_id 是否正确
3. 手动执行激活SQL（步骤5）

### 问题2: 管理员账号未创建

**原因**: 激活逻辑中创建管理员的代码未执行

**解决方案**:
1. 检查是否已存在管理员账号
2. 手动执行创建管理员SQL（步骤5.4）

### 问题3: 授权码未生成

**原因**: 授权码生成逻辑未执行

**解决方案**:
1. 手动生成授权码（步骤5.1）
2. 更新租户表（步骤5.2）

### 问题4: 无法登录系统

**可能原因**:
1. 租户状态不是 'active'
2. 授权码不正确
3. 管理员账号未创建
4. 密码不正确

**解决方案**:
1. 检查租户状态
2. 检查授权码是否正确
3. 检查管理员账号是否存在
4. 确认密码是 admin123
