## 完整示例：注册流程

```typescript
import { aliyunSmsService } from '../services/AliyunSmsService'

async function handleRegistration(phone: string, tenantData: any) {
  try {
    // 1. 发送验证码
    const verifyResult = await aliyunSmsService.sendVerificationCode(phone, '123456')
    if (!verifyResult.success) {
      throw new Error('验证码发送失败')
    }

    // 2. 用户输入验证码并验证...
    // ... 验证逻辑 ...

    // 3. 创建租户账号
    const tenant = await createTenant(tenantData)

    // 4. 发送注册成功通知
    const notifyResult = await aliyunSmsService.sendSms(phone, 'REGISTER_SUCCESS', {
      tenant_name: tenant.name,
      license_code: tenant.licenseCode,
      admin_account: tenant.adminAccount,
      admin_password: tenant.adminPassword
    })

    if (!notifyResult.success) {
      console.error('注册成功通知发送失败:', notifyResult.message)
      // 注意：这里不应该阻断注册流程，只记录日志
    }

    return { success: true, tenant }
  } catch (error) {
    console.error('注册失败:', error)
    return { success: false, message: error.message }
  }
}
```

async function handlePaymentSuccess(orderId: string) {
  try {
    // 1. 查询订单信息
    const order = await getOrderById(orderId)

    console.error('支付回调处理失败:', error)
    return { success: false, message: error.message }
  }

## 完整示例：定时任务 - 到期提醒

```typescript
import { aliyunSmsService } from '../services/AliyunSmsService'
import { AppDataSource } from '../config/database'

async function sendExpirationReminders() {
  console.log('[定时任务] 开始检查即将到期的租户...')

  try {
    // 查询7天、3天、1天后到期的租户
    const tenants = await AppDataSource.query(`
      SELECT 
        t.id,
        t.name as tenant_name,
        t.contact_phone,
        DATEDIFF(t.expire_date, NOW()) as days_left
      FROM tenants t
      WHERE t.status = 'active'
        AND t.expire_date > NOW()
        AND DATEDIFF(t.expire_date, NOW()) IN (7, 3, 1)
        AND (
          t.last_remind_date IS NULL 
          OR DATE(t.last_remind_date) < CURDATE()
        )
    `)

    console.log(`找到 ${tenants.length} 个需要提醒的租户`)

    for (const tenant of tenants) {
      try {
        const result = await aliyunSmsService.sendSms(
          tenant.contact_phone,
          'EXPIRE_REMIND',
          {
            tenant_name: tenant.tenant_name,
            days_left: String(tenant.days_left)
          }
        )

        if (result.success) {
          // 更新最后提醒时间
          await AppDataSource.query(
            `UPDATE tenants SET last_remind_date = NOW() WHERE id = ?`,
            [tenant.id]
          )
          console.log(`✅ 已提醒租户: ${tenant.tenant_name} (剩余${tenant.days_left}天)`)
        } else {
          console.error(`❌ 提醒失败: ${tenant.tenant_name} - ${result.message}`)
        }
      } catch (error) {
        console.error(`处理租户 ${tenant.tenant_name} 时出错:`, error)
      }
    }

    console.log('[定时任务] 到期提醒任务完成')
  } catch (error) {
    console.error('[定时任务] 到期提醒任务失败:', error)
  }
}

// 每天早上9点执行
// 在 SchedulerService 中注册此任务
```

async function sendExpiredNotices() {
  console.log('[定时任务] 开始检查已过期的租户...')

  try {
    // 查询今天过期的租户（尚未发送过期通知）
    const tenants = await AppDataSource.query(`
      SELECT 
        t.id,
        t.name as tenant_name,
        t.contact_phone
      FROM tenants t
      WHERE t.status = 'active'
        AND DATE(t.expire_date) = CURDATE()
        AND (t.expired_notice_sent IS NULL OR t.expired_notice_sent = 0)
    `)

    console.log(`找到 ${tenants.length} 个今天过期的租户`)

    for (const tenant of tenants) {

        const result = await aliyunSmsService.sendSms(
          tenant.contact_phone,

          {
            tenant_name: tenant.tenant_name
          }

          // 标记已发送过期通知

            `UPDATE tenants SET expired_notice_sent = 1, status = 'expired' WHERE id = ?`,
            [tenant.id]

        console.error(`处理租户 ${tenant.tenant_name} 时出错:`, error)

  } catch (error) {
    console.error('[定时任务] 过期通知任务失败:', error)
  }

## 错误处理最佳实践

```typescript
async function sendSmsWithRetry(phone: string, type: string, params: any, maxRetries = 3) {
  let lastError = null

  for (let i = 0; i < maxRetries; i++) {
    try {
      const result = await aliyunSmsService.sendSms(phone, type, params)
      
      if (result.success) {
        return { success: true }
      }

      lastError = result.message

      // 某些错误不需要重试
      if (result.message?.includes('手机号格式错误') || 
          result.message?.includes('签名不合法')) {
        break
      }

      // 等待后重试
      if (i < maxRetries - 1) {
        await new Promise(resolve => setTimeout(resolve, 1000 * (i + 1)))
      }
    } catch (error) {
      lastError = error.message
    }
  }

  return { success: false, message: lastError }
}
```

## 注意事项

1. **频率限制**：阿里云对同一手机号有发送频率限制（通常1分钟1条），注意控制发送频率

2. **错误处理**：短信发送失败不应阻断主业务流程，应该记录日志并考虑重试机制

3. **模板变量**：确保传递的变量名称与阿里云模板中定义的变量名完全一致

4. **测试环境**：在测试环境中，如果未配置阿里云短信，服务会自动进入开发模式，将内容打印到控制台

5. **成本控制**：短信是按条收费的，建议：
   - 添加发送日志和统计
   - 设置每日发送上限
   - 对重要通知才发送短信，一般通知可以用站内信

6. **隐私保护**：日志中不要记录完整的手机号和敏感信息

## 配置检查

在使用前，可以检查配置是否完整：

```typescript
async function checkSmsConfig() {
  const loaded = await aliyunSmsService.loadFromDatabase()
  
  if (!loaded) {
    console.warn('短信服务未配置或未启用')
    return false
  }

  console.log('短信服务配置正常')
  return true
}
```
