**操作类型映射**:
| 操作类型 | 显示文本 | 标签颜色 |
|---------|---------|---------|
| create | 创建租户 | success |
| update | 更新信息 | primary |
| delete | 删除租户 | danger |
| suspend | 暂停授权 | warning |
| resume | 恢复授权 | success |
| enable | 启用租户 | success |
| disable | 禁用租户 | info |
| renew | 续期 | primary |
| regenerate_license | 重新生成授权码 | warning |
| adjust_quota | 调整配额 | primary |
| adjust_package | 调整套餐 | primary |

**核心代码**:
```javascript
const fetchLogs = async () => {
  const res = await request.get(`/tenants/${route.params.id}/logs`, {
    params: { page: logPage.value, pageSize: logPageSize.value }
  })
  logs.value = res.data.logs || []
  logTotal.value = res.data.pagination?.total || 0
}

const getActionType = (action: string): string => {
  const map = {
    create: 'success', update: 'primary', delete: 'danger',
    suspend: 'warning', resume: 'success', enable: 'success',
    disable: 'info', renew: 'primary', regenerate_license: 'warning',
    adjust_quota: 'primary', adjust_package: 'primary'
  }
  return map[action] || 'info'
}

const formatLogDetails = (details: any): string => {
  // 格式化操作详情，显示变更内容
  if (details.changes) {
    return Object.entries(details.changes)
      .map(([key, value]) => `${key}: ${value.old} → ${value.new}`)
      .join(', ')
  }
  return JSON.stringify(details)
}
```
