## 技术要点

### 1. Admin权限验证

```typescript
// Admin用户不受租户限制
if (user.role !== 'admin' && user.role !== 'super_admin') {
  return res.status(403).json({
    success: false,
    message: '权限不足，仅管理员可访问'
  });
}
```

### 2. 租户数据查询（Admin视角）

```typescript
// Admin可以查询所有租户的数据
// 不使用BaseRepository，直接使用原生Repository
const tenantRepo = AppDataSource.getRepository(Tenant);
const tenants = await tenantRepo.find({
  where: { status: 'active' }
  // 不添加tenant_id过滤
});
```

### 3. 授权码生成

```typescript
// 使用UUID生成唯一授权码
const licenseKey = Tenant.generateLicenseKey();
// 格式: LIC-XXXXXXXXXXXXXXXXXXXX (20位大写字母数字)
```

### 4. 软删除

```typescript
// 不真正删除数据，只标记为inactive
tenant.status = 'inactive';
await tenantRepo.save(tenant);
```

### 5. 统计数据

```typescript
// 查询租户的用户数
const userCount = await userRepo.count({
  where: { tenantId: id }
});

// 计算使用率
const userUsagePercent = (userCount / tenant.maxUsers) * 100;
```
