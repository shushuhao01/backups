#### 步骤 1：创建 Tenant 实体文件
**文件路径**：`backend/src/entities/Tenant.ts`

```typescript
import { Entity, PrimaryColumn, Column, CreateDateColumn, UpdateDateColumn, Index } from 'typeorm';
import { v4 as uuidv4 } from 'uuid';

@Entity('tenants')
export class Tenant {
  @PrimaryColumn('varchar', { length: 36 })
  id: string;

  @Column('varchar', { length: 100 })
  name: string;

  @Index('idx_tenant_code', { unique: true })
  @Column('varchar', { length: 50, unique: true })
  code: string;

  @Column('enum', { enum: ['active', 'inactive', 'expired'], default: 'active' })
  status: 'active' | 'inactive' | 'expired';

  @Index('idx_license_key', { unique: true })
  @Column('varchar', { length: 100, unique: true, name: 'license_key' })
  licenseKey: string;

  @Column('datetime', { nullable: true, name: 'expire_date' })
  expireDate: Date | null;

  @Column('int', { default: 10, name: 'max_users' })
  maxUsers: number;

  @Column('int', { default: 10, name: 'max_storage_gb' })
  maxStorageGb: number;

  @Column('varchar', { length: 50, nullable: true, name: 'contact_name' })
  contactName: string | null;

  @Column('varchar', { length: 20, nullable: true, name: 'contact_phone' })
  contactPhone: string | null;

  @Column('varchar', { length: 100, nullable: true, name: 'contact_email' })
  contactEmail: string | null;

  @CreateDateColumn({ name: 'created_at' })
  createdAt: Date;

  @UpdateDateColumn({ name: 'updated_at' })
  updatedAt: Date;

  // 生成授权码
  static generateLicenseKey(): string {
    return `LIC-${uuidv4().replace(/-/g, '').substring(0, 20).toUpperCase()}`;
  }

  // 检查是否过期
  isExpired(): boolean {
    if (!this.expireDate) return false;
    return new Date(this.expireDate) < new Date();
  }

  // 检查是否可用
  isAvailable(): boolean {
    return this.status === 'active' && !this.isExpired();
  }
}
```

#### 步骤 2：创建数据库迁移 SQL
**文件路径**：`backend/database-migrations/create-tenants-table.sql`

```sql
-- 创建租户表
CREATE TABLE IF NOT EXISTS `tenants` (
  `id` VARCHAR(36) PRIMARY KEY COMMENT '租户ID（UUID）',
  `name` VARCHAR(100) NOT NULL COMMENT '租户名称',
  `code` VARCHAR(50) NOT NULL UNIQUE COMMENT '租户代码',
  `status` ENUM('active', 'inactive', 'expired') DEFAULT 'active' COMMENT '状态',
  `license_key` VARCHAR(100) NOT NULL UNIQUE COMMENT '授权码',
  `expire_date` DATETIME NULL COMMENT '过期时间',
  `max_users` INT DEFAULT 10 COMMENT '最大用户数',
  `max_storage_gb` INT DEFAULT 10 COMMENT '最大存储空间（GB）',
  `contact_name` VARCHAR(50) NULL COMMENT '联系人',
  `contact_phone` VARCHAR(20) NULL COMMENT '联系电话',
  `contact_email` VARCHAR(100) NULL COMMENT '联系邮箱',
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  INDEX `idx_tenant_code` (`code`),
  INDEX `idx_license_key` (`license_key`),
  INDEX `idx_status` (`status`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='租户表';
```
