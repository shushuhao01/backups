## 🔧 阶段1: 后端核心功能开发

### 1.1 创建统一工具函数

**文件**: `backend/src/utils/adminAccountHelper.ts`（新建）

```typescript
import { AppDataSource } from '../config/database';
import { v4 as uuidv4 } from 'uuid';
import * as crypto from 'crypto';

/**
 * 生成默认密码
 */
export const generateDefaultPassword = (): string => {
  return 'Aa123456';  // 符合复杂度：大写+小写+数字
};

/**
 * 密码加密
 */
export const hashPassword = (password: string): { hash: string; salt: string } => {
  const salt = crypto.randomBytes(16).toString('hex');
  const hash = crypto.pbkdf2Sync(password, salt, 1000, 64, 'sha512').toString('hex');
  return { hash, salt };
};

/**
 * 创建默认管理员账号
 * @param params 参数
 * @returns 账号密码信息
 */
export const createDefaultAdmin = async (params: {
  tenantId: string | null;  // SaaS租户ID，私有部署为null
  phone: string;            // 手机号（作为用户名）
  realName: string;         // 真实姓名
  email?: string;           // 邮箱（可选）
}): Promise<{ username: string; password: string }> => {
  const { tenantId, phone, realName, email } = params;

  // 检查用户名是否已存在
  const existing = await AppDataSource.query(
    'SELECT id FROM users WHERE username = ?',
    [phone]
  );

  if (existing.length > 0) {
    console.warn(`[AdminAccountHelper] 用户名 ${phone} 已存在，跳过创建`);
    return { username: phone, password: generateDefaultPassword() };
  }

  const userId = uuidv4();
  const password = generateDefaultPassword();
  const { hash, salt } = hashPassword(password);
  const now = new Date().toISOString().slice(0, 19).replace('T', ' ');

  await AppDataSource.query(
    `INSERT INTO users (
      id, tenant_id, username, password, salt, real_name, phone, email,
      role, status, created_at, updated_at
    ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, 'admin', 'active', ?, ?)`,
    [userId, tenantId, phone, hash, salt, realName, phone, email || null, now, now]
  );

  console.log(`[AdminAccountHelper] 已创建管理员账号: ${phone}`);
  return { username: phone, password };
};
```
