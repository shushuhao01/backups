```typescript
// ✅ 支付成功后生成 TENANT- 前缀授权码
private generateLicenseKey(): string {
  const segments = []
  for (let i = 0; i < 4; i++) {
    segments.push(crypto.randomBytes(2).toString('hex').toUpperCase())
  }
  return `TENANT-${segments.join('-')}`  // 格式: TENANT-ABCD-EFGH-IJKL-MNOP
}

// 激活租户
private async activateTenant(tenantId: string, packageId: string): Promise<string | null> {
  const licenseKey = this.generateLicenseKey()
  await AppDataSource.query(
    `UPDATE tenants SET license_key = ?, license_status = 'active', 
     expire_date = ?, max_users = ? WHERE id = ?`,
    [licenseKey, expireDate, maxUsers, tenantId]
  )
  return licenseKey
}
```

**特点**:
- ✅ 使用 `TENANT-` 前缀，符合SaaS租户规范
- ✅ 支付成功后自动更新租户表的授权码
- ✅ 同时更新授权状态为 `active`

```typescript
// ✅ 支付查询接口返回授权码和租户编码
router.get('/query/:orderNo', async (req: Request, res: Response) => {
  const order = await paymentService.queryOrder(orderNo)
  
  // 如果已支付，查询租户的授权码
  let licenseKey = null
  let tenantCode = null
  if (order.status === 'paid' && order.tenant_id) {
    const tenants = await AppDataSource.query(
      'SELECT code, license_key FROM tenants WHERE id = ?', [order.tenant_id]
    )
    if (tenants.length > 0) {
      tenantCode = tenants[0].code
      licenseKey = tenants[0].license_key
    }
  }
  
  res.json({
    code: 0,
    data: {
      orderNo, status, amount, payType, paidAt,
      tenantId: order.tenant_id,
      tenantCode,      // ✅ 返回租户编码
      licenseKey       // ✅ 返回授权码
    }
  })
})
```

**特点**:
- ✅ 支付成功后，前端轮询此接口获取授权码

## 6️⃣ 问题汇总与修复建议

### 问题1: Admin后台授权码缺少PRIVATE-前缀

**当前代码**: `backend/src/routes/admin/licenses.ts`
```typescript
// ❌ 错误：缺少前缀
const generateLicenseKey = (): string => {
  const segments = [];
  for (let i = 0; i < 4; i++) {
    segments.push(crypto.randomBytes(4).toString('hex').toUpperCase());
  }
  return segments.join('-');
};
```

**修复建议**:
```typescript
// ✅ 正确：添加PRIVATE-前缀
const generateLicenseKey = (): string => {
  const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';
  const segments = [];
  for (let i = 0; i < 4; i++) {
    let segment = '';
    for (let j = 0; j < 4; j++) {
      segment += chars[Math.floor(Math.random() * chars.length)];
    }
    segments.push(segment);
  }
  return `PRIVATE-${segments.join('-')}`;
};
```


### 问题2: Admin后台不支持生成SaaS租户授权码

**现状**: Admin后台只能生成私有部署授权码，无法为SaaS租户手动生成授权码

**影响**: 
- 无法为特殊客户手动创建SaaS租户授权码
- 所有SaaS租户必须通过官网注册支付流程

**修复建议**: 
在Admin后台授权码生成页面添加"授权码类型"选项：
```typescript
// 添加授权码类型选择
<el-form-item label="授权码类型" prop="licenseCodeType">
  <el-radio-group v-model="form.licenseCodeType">
    <el-radio label="private">私有部署（PRIVATE-前缀）</el-radio>
    <el-radio label="saas">SaaS租户（TENANT-前缀）</el-radio>
  </el-radio-group>
</el-form-item>

// 后端根据类型生成不同前缀的授权码
const generateLicenseKey = (type: 'private' | 'saas'): string => {
  const prefix = type === 'private' ? 'PRIVATE' : 'TENANT';
  // ... 生成逻辑
  return `${prefix}-${segments.join('-')}`;
};
```
